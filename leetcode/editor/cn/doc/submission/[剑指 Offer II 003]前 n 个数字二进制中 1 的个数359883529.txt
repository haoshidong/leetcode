class Solution {
    public int[] countBits(int n) {
        int[] res = new int[n+1];
        for (int i=0;i<=n;i++){
            int k = 0;
            int value = i;
            while (value>0){
                k += (value%2);
                value /= 2;
            }
            res[i] = k;
        }
        return res;
    }
}
