class Solution {
    public int[] countBits(int n) {
        int[] res = new int[n];
        for (int i=0;i<=n;i++){
            int k = 0;
            while (i>0){
                k += (i%2);
                i /= 2;
            }
            res[i] = k;
        }
        return res;
    }
}
